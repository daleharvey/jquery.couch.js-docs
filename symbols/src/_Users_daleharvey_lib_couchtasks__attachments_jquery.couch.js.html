 45   function encodeDocId(docID) {
 46     var parts = docID.split("/");
 47     if (parts[0] == "_design") {
 48       parts.shift();
 49       return "_design/" + encodeURIComponent(parts.join('/'));
 50     }
 51     return encodeURIComponent(docID);
 52   }

 54   /**
 55    * @private
 56    */
 57   function prepareUserDoc(user_doc, new_password) {
 58     if (typeof hex_sha1 == "undefined") {
 59       alert("creating a user doc requires sha1.js to be loaded in the page");
 60       return;
 61     }
 62     var user_prefix = "org.couchdb.user:";
 63     user_doc._id = user_doc._id || user_prefix + user_doc.name;
 64     if (new_password) {
 65       // handle the password crypto
 66       user_doc.salt = $.couch.newUUID();
 67       user_doc.password_sha = hex_sha1(new_password + user_doc.salt);
 68     }
 69     user_doc.type = "user";
 70     if (!user_doc.roles) {
 71       user_doc.roles = [];
 72     }
 73     return user_doc;
 74   }

416         /**
417          * @namespace
418          * $.couch.db.changes provides an API for subscribing to the changes
419          * feed
420          * <pre><code>var $changes = $.couch.db("mydatabase").changes();
421          *$changes.onChange = function (data) {
422          *    ... process data ...
423          * }
424          * $changes.stop();
425          * </code></pre>
426          */
427         changes: function(since, options) {
428 
429           options = options || {};
430           // set up the promise object within a closure for this handler
431           var timeout = 100, db = this, active = true,
432             listeners = [],
433             promise = /** @lends $.couch.db.changes */ {
434               /**
435                * Add a listener callback
436                * @see <a href="http://techzone.couchbase.com/sites/default/
437                * files/uploads/all/documentation/couchbase-api-db.html#couch
438                * base-api-db_db-changes_get">docs for /db/_changes</a>
439                * @param {Function} fun Callback function to run when
440                * notified of changes.
441                */
442               onChange : function(fun) {
443                 listeners.push(fun);
444               },
445               /**
446                * Stop subscribing to the changes feed
447                */
448               stop : function() {
449                 active = false;
450               }
451             };
452           // call each listener when there is a change
453           function triggerListeners(resp) {
454             $.each(listeners, function() {
455               this(resp);
456             });
457           };
458           // when there is a change, call any listeners, then check for
459           // another change
460           options.success = function(resp) {
461             timeout = 100;
462             if (active) {
463               since = resp.last_seq;
464               triggerListeners(resp);
465               getChangesSince();
466             };
467           };
468           options.error = function() {
469             if (active) {
470               setTimeout(getChangesSince, timeout);
471               timeout = timeout * 2;
472             }
473           };
474           // actually make the changes request
475           function getChangesSince() {
476             var opts = $.extend({heartbeat : 10 * 1000}, options, {
477               feed : "longpoll",
478               since : since
479             });
480             ajax(
481               {url: db.uri + "_changes"+encodeOptions(opts)},
482               options,
483               "Error connecting to "+db.uri+"/_changes."
484             );
485           }
486           // start the first request
487           if (since) {
488             getChangesSince();
489           } else {
490             db.info({
491               success : function(info) {
492                 since = info.update_seq;
493                 getChangesSince();
494               }
495             });
496           }
497           return promise;
498         },

618         /**
619          * Create a new document in the specified database, using the supplied
620          * JSON document structure. If the JSON structure includes the _id
621          * field, then the document will be created with the specified document
622          * ID. If the _id field is not specified, a new unique ID will be
623          * generated.
624          * @see <a href="http://techzone.couchbase.com/sites/default/files/
625          * uploads/all/documentation/couchbase-api-dbdoc.html#couchbase-api-
626          * dbdoc_db_post">docs for GET /db/doc</a>
627          * @param {String} doc document to save
628          * @param {ajaxSettings} options <a href="http://api.jquery.com/
629          * jQuery.ajax/#jQuery-ajax-settings">jQuery ajax settings</a>
630          */
631         saveDoc: function(doc, options) {
632           options = options || {};
633           var db = this;
634           var beforeSend = fullCommit(options);
635           if (doc._id === undefined) {
636             var method = "POST";
637             var uri = this.uri;
638           } else {
639             var method = "PUT";
640             var uri = this.uri + encodeDocId(doc._id);
641           }
642           var versioned = maybeApplyVersion(doc);
643           $.ajax({
644             type: method, url: uri + encodeOptions(options),
645             contentType: "application/json",
646             dataType: "json", data: toJSON(doc),
647             beforeSend : beforeSend,
648             complete: function(req) {
649               var resp = httpData(req, "json");
650               if (req.status == 200 || req.status == 201 || req.status == 202) {
651                 doc._id = resp.id;
652                 doc._rev = resp.rev;
653                 if (versioned) {
654                   db.openDoc(doc._id, {
655                     attachPrevRev : true,
656                     success : function(d) {
657                       doc._attachments = d._attachments;
658                       if (options.success) options.success(resp);
659                     }
660                   });
661                 } else {
662                   if (options.success) options.success(resp);
663                 }
664               } else if (options.error) {
665                 options.error(req.status, resp.error, resp.reason);
666               } else {
667                 alert("The document could not be saved: " + resp.reason);
668               }
669             }

997     var ct = xhr.getResponseHeader("content-type") || "",
998       xml = type === "xml" || !type && ct.indexOf("xml") >= 0,

1021   function ajax(obj, options, errorMessage, ajaxOptions) {
1022     options = $.extend({successStatus: 200}, options);
1023     ajaxOptions = $.extend({contentType: "application/json"}, ajaxOptions);
1024     errorMessage = errorMessage || "Unknown error";
1025     $.ajax($.extend($.extend({
1026       type: "GET", dataType: "json", cache : !$.browser.msie,
1027       beforeSend: function(xhr){
1028         if(ajaxOptions && ajaxOptions.headers){
1029           for (var header in ajaxOptions.headers){
1030             xhr.setRequestHeader(header, ajaxOptions.headers[header]);
1031           }
1032         }
1033       },
1034       complete: function(req) {
1035         try {
1036           var resp = httpData(req, "json");
1037         } catch(e) {
1038           if (options.error) {
1039             options.error(req.status, req, e);
1040           } else {
1041             alert(errorMessage + ": " + e);
1042           }
1043           return;
1044         }
1045         if (options.ajaxStart) {
1046           options.ajaxStart(resp);
1047         }
1048         if (req.status == options.successStatus) {
1049           if (options.beforeSuccess) options.beforeSuccess(req, resp);
1050           if (options.success) options.success(resp);
1051         } else if (options.error) {
1052           options.error(req.status, resp && resp.error ||
1053                         errorMessage, resp && resp.reason || "no response");
1054         } else {
1055           alert(errorMessage + ": " + resp.reason);
1056         }
1057       }
1058     }, obj), ajaxOptions));
1059   }

1079   // Convert a options object to an url query string.
1080   // ex: {key:'value',key2:'value2'} becomes '?key="value"&key2="value2"'
1081   function encodeOptions(options) {
1082     var buf = [];
1083     if (typeof(options) === "object" && options !== null) {
1084       for (var name in options) {
1085         if ($.inArray(name,
1086                       ["error", "success", "beforeSuccess", "ajaxStart"]) >= 0)
1087           continue;
1088         var value = options[name];
1089         if ($.inArray(name, ["key", "startkey", "endkey"]) >= 0) {
1090           value = toJSON(value);
1091         }
1092         buf.push(encodeURIComponent(name) + "=" + encodeURIComponent(value));
1093       }
1094     }
1095     return buf.length ? "?" + buf.join("&") : "";
1096   }
